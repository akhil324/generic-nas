# Stop the current container:
docker stop rpi-tester

# Rebuild the image:
docker build -t rpi-setup-test -f Dockerfile.rpi_test_env .

Run Container:
docker run --name rpi-tester --rm -it --privileged \
    --volume=/sys/fs/cgroup:/sys/fs/cgroup \
    --publish 2222:22 \
    --publish 8080:8080 \
    --publish 8081:8081 \
    rpi-setup-test

Access the Container via SSH:
ssh gnas@localhost -p 2222

Test Individual Scripts
sudo su -
cd /boot/firmware/
bash ./setup_scripts/setup_autologin.sh
bash ./setup_scripts/setup_vpn.sh
bash ./setup_scripts/manage_storage.sh
bash ./setup_scripts/setup_filebrowser.sh
bash ./setup_scripts/setup_monitoring.sh
